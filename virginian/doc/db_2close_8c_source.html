00005  * @brief Close the currently opened database
00006  *
00007  * Close the open database. This is accomplished by clearing every single
00008  * main-memory tablet slot, thus ensuring the changes to every tablet are
00009  * reflected on disk, then writing the fixed-size virg_db struct stored in the
00010  * virginian state struct to the head of the database file, then writing the
00011  * variable-sized meta information between this fixed-size area and the start of
00012  * the actual tablets. The variable-size meta information is a list of all the
00013  * tablets in the file. This function should be called only if no tablets are
00014  * locked in memory.
00015  *
00016  * @param v Pointer to the state struct of the database system
00017  * @return VIRG_SUCCESS or VIRG_FAIL depending on errors during the function
00018  * call
00019  */
00020 int virg_db_close(virginian *v) {
00021         unsigned i, r;
00022         size_t size;
00023 
00024         // check that a database is currently open
00025         if(v->dbfd == -1)
00026                 return VIRG_SUCCESS;
00027 
00028         // set size to the meta area of the file
00029         size = sizeof(virg_db);
00030         size += v->db.alloced_tablets * sizeof(virg_tablet_info);
00031 
00032         // clear every tablet slot, thus writing every in-memory tablet to disk
00033         for(i = 0; i < VIRG_MEM_TABLETS; i++)
00034                 if(v->tablet_slot_status[i])
00035                         virg_db_clear(v, i);
00036 
00037         // write the fixed-size meta information in the virg_db struct to disk
00038         r = pwrite(v->dbfd, &v->db, sizeof(virg_db), 0);
00039         VIRG_CHECK(r < sizeof(virg_db), "Problem writing db meta info");
00040 
00041         // write the variable-sized meta information (the tablet slot use and ids)
00042         // to disk
00043         size = v->db.alloced_tablets * sizeof(virg_tablet_info);
00044         r = pwrite(v->dbfd, v->db.tablet_info, size, sizeof(virg_db));
00045         VIRG_CHECK(r < size, "Problem writing db meta info");
00046 
00047         // close the database file
00048         r = close(v->dbfd);
00049         VIRG_CHECK(r != 0, "Problem closing file");
00050         v->dbfd = -1;
00051 
00052         // free the variable size meta information block
00053         free(v->db.tablet_info);
00054
