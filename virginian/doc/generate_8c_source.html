00050 /** Compares two expressions, recursing to sub-expressions if necessary, to
00051  * determine if they are identical. This is used in register assignment to check
00052  * if an expression has already been stored in a register.
00053  */
00054 int expr_equal(node_expr *x1, node_expr *x2)
00055 {
00056         if(x1->type != x2->type)
00057                 return 0;
00058 
00059         switch(x1->type) {
00060                 case NODE_EXPR_INT :
00061                         return x1->val.i == x2->val.i;
00062 
00063                 case NODE_EXPR_FLOAT :
00064                         return x1->val.f == x2->val.f;
00065 
00066                 case NODE_EXPR_STRING :
00067                         return strcmp(x1->val.s, x2->val.s) == 0;
00068 
00069                 // if both are columns, we compare the column id
00070                 case NODE_EXPR_COLUMN :
00071                         return (x1->iskey == x2->iskey && x1->val.i == x2->val.i);
00072 
00073                 // if this is a math op we must compare the op and both sides
00074                 case NODE_EXPR_OP :
00075                         return x1->val.i == x2->val.i &&
00076                                 expr_equal(x1->lhs, x2->lhs) &&
00077                                 expr_equal(x1->rhs, x2->rhs);
00078 
00079                 default :
00080                         assert(0);
00081         }
00082 }

00293  * are constants. If that is the case, perform the math necessary to simplyify
00294  * the operation to a single constant
00295  */
00296 void select_resolveopspass_recurse(node_expr *x)

00299         if(x->type != NODE_EXPR_OP)
00300                 return;

00467  * filtering results. At a high level, this works in the following manner. Each
00468  * condition is an operator that compares two expressions with a boolean result.
00469  * Conditions can also have AND and OR conditions, thus forming a tree. If a
00470  * condition node has an AND condition, we know that failure of this node's
00471  * condition invalidates the clause, thus we jump forward to the next condition
00472  * to be evaluated, possibly the end of the conditions. If this node has no
00473  * children or is group with an OR condition at a higher order of operations

00628         if(root->conditions != NULL) {
00629                 absop *stub = create_absop(OP_Nop, 0, 0, 0, NULL);
00630 
00631                 // recurse through condition tree, adding opcodes as necessary
00632                 select_structurepass_condrecurse(root->conditions, ops_list,
00633                         stub, result, NULL);
00634 
00635                 // since conditions jump forward to output results, anything that doesnt
00636                 // jump is assumed to be invalid
00637                 newop = create_absop(OP_Invalid, 0, 0, 0, NULL);
00638                 append(&ops_list, newop);
00639 
00640                 append(&ops_list, stub);
00641         }
00642 
00643         // resolve result column expressions
00644         currcol = root->resultcols;
00645         for(; currcol != NULL; currcol = currcol->next) {
00646                 int reg = select_structurepass_expr(ops_list, currcol->expr);
00647                 currcol->output_reg = reg;
00648         }
00649 
00650         // rearrange registers so output columns are contiguous
00651         // TODO check to make sure resultcols dont use the same reg
00652         //              copy if they do
00653         regindex();
00654         currcol = root->resultcols;
00655         int numrescols = 0;
00656 
00657         for(; currcol != NULL; currcol = currcol->next) {
00658                 numrescols++;
00659                 int i = currcol->output_reg;
00660 
00661                 // already at back of registers
00662                 if(reg_table[i].index == regcounter - 1)
00663                         continue;
00664 
00665                 int oldindex = reg_table[i].index;
00666                 int oldi = i;
00667                 reg_table[i].index = regcounter - 1;
00668 
00669                 //printf("column %i old %i new %i\n", numrescols - 1, oldindex, regcounter - 1);
00670 
00671                 for(i = 0; i < regcounter; i++)
00672                         if(i != oldi && reg_table[i].index > oldindex)
00673                                 reg_table[i].index--;
00674         }
00675 
00676         // output result columns
00677         result->op.p1 = root->resultcols->output_reg;
00678         result->op.p2 = numrescols;
00679         append(&ops_list, result);
00680 
00681         // finish up parallel section
00682         append(&ops_list, converge);
00683 
00684         // add finish op
00685         newop = create_absop(OP_Finish, 0, 0, 0, NULL);
00686         append(&ops_list, newop);
00687 
00688         ops[0] = ops_list;
00689         return VIRG_SUCCESS;

00705 /** Pass 4
00706  * This pass resolves the actual location of registers, since they've been
00707  * rearranged in the original array, and also handles cases where an op jumps to
00708  * another op. We must replace the pointer with the index to that other op
00709  */
00710 void select_registerpass(absop *aop)
00711 {
00712         // iterate through all ops
00713         for(; aop != NULL; aop = aop->next) {
00714                 //fprintf(stderr, "::%i\n", aop->op.op);
00715                 switch(aop->op.op) {
00716                         case OP_Parallel :
00717                                 // resolve index of forward op we are pointing to with opptr
00718                                 aop->op.p3 = aop->opptr->index + 1;
00719                                 break;
00720 
00721                         case OP_Integer :
00722                         case OP_Column :
00723                         case OP_Rowid :
00724                         case OP_Result :
00725                         case OP_Float :
00726                                 // resolve actual register index
00727                                 aop->op.p1 = reg_table[aop->op.p1].index;
00728                                 break;
00729 
00730                         // these ops resolve 3 registers
00731                         case OP_Add :
00732                         case OP_Sub :
00733                         case OP_Mul :
00734                         case OP_Div :
00735                                 aop->op.p1 = reg_table[aop->op.p1].index;
00736                                 aop->op.p2 = reg_table[aop->op.p2].index;
00737                                 aop->op.p3 = reg_table[aop->op.p3].index;
00738                                 break;
00739 
00740                         // resolve 2 registers and forward pointing jump location
00741                         case OP_Eq :
00742                         case OP_Neq :
00743                         case OP_Le :
00744                         case OP_Lt :
00745                         case OP_Ge :
00746                         case OP_Gt :
00747                                 aop->op.p1 = reg_table[aop->op.p1].index;
00748                                 aop->op.p2 = reg_table[aop->op.p2].index;
00749                                 aop->op.p3 = aop->opptr->index;
00750                                 break;
00751 
00752                         // no register use
00753                         case OP_Table :
00754                         case OP_Invalid :
00755                         case OP_ResultColumn :
00756                         case OP_Converge :
00757                         case OP_Finish :
00758                         case OP_Nop :
00759                                 break;
00760 
00761                         default :
00762                                 assert(0);
00763                 }
00764         }
00765 }
