00006  *
00007  * Expands the fixed-size block of the tablet by moving the variable block back
00008  * farther. This is a convenience function for other tablet functions, and
00009  * leaves the tablet inconsistent because no virg_tablet_meta.possible_row or
00010  * column information is edited, so it should be followed up with these changes
00011  * to the tablet as well. Note that if the variable size block has a size of
00012  * zero, this is very easy, as we just move its location back in the meta
00013  * information. If not, however, we must perform a safe memory copy to transfer
00014  * its contents farther back in the tablet.
00015  *
00016  * @param tab   Pointer to the tablet to be changed
00017  * @param size  Additional size to be added to the fixed block
00018  * @return VIRG_SUCCESS or VIRG_FAIL depending on errors during the function
00019  * call
00020  */
00021 int virg_tablet_growfixed(virg_tablet_meta *tab, size_t size)
00022 {
00023         // make sure you can add that much to the tablet
00024         VIRG_CHECK(tab->size + size > VIRG_TABLET_SIZE, "Too big to grow, the tablet must be split");
00025 
00026         // if the variable block has a zero size, we just move it back and return
00027         // otherwise we have to copy its contents
00028         if(tab->size == tab->variable_block) {
00029                 tab->variable_block += size;
00030                 tab->size += size;
00031                 return VIRG_SUCCESS;
00032         }
00033 
00034         size_t new_variable = tab->variable_block + size;
00035         size_t variable_size = tab->size - tab->variable_block;
00036 
00037         char *dest = (char*)tab + new_variable;
00038         char *src = (char*)tab + tab->variable_block;
00039 
00040         // move variable-sized information
00041         memmove(dest, src, variable_size);
00042 
00043         tab->variable_block = new_variable;
00044         tab->size = new_variable + variable_size;
00045 
00046         return VIRG_SUCCESS;
00047 }
