00004  * @ingroup database
00005  * @brief Load a tablet into a slot based on its ID and return a pointer
00006  *
00007  * This function is used when you are attempting to access a tablet based on its
00008  * ID. It will add a read-lock for the tablet and return a pointer to it. For
00009  * efficiency, it first checks to see if the tablet already resides in a
00010  * main-memory tablet slot. If it does, we simple add the lock and return the
00011  * pointer. Otherwise we must fetch the tablet from the database file on disk
00012  * and read it into a tablet slot. This function is thread-safe and performs
00013  * several checks to ensure that the tablet ID actually exists and that the
00014  * expected size of the tablet is actually read.
00015  *
00016  * @param v Pointer to the state struct of the database system
00017  * @param tablet_id The ID of the tablet being loaded
00018  * @param tab A pointer to a tablet pointer through which the location of the loaded tablet is returned
00019  * @return VIRG_SUCCESS or VIRG_FAIL depending on errors during the function
00020  * call
00021  */

00028         pthread_mutex_lock(&v->slot_lock);
00029 
00030         // check if the tablet is already loaded
00031         for(i = 0; i < VIRG_MEM_TABLETS; i++)
00032                 if(v->tablet_slot_status[i] != 0 && v->tablet_slot_ids[i] == tablet_id) {
00033                         if(tab != NULL)
00034                                 tab[0] = v->tablet_slots[i];
00035 
00036                         // if so, add an extra tablet lock and return the pointer
00037                         v->tablet_slot_status[i]++;

00039                         return VIRG_SUCCESS;
00040                 }

00052         // the tablet should never not be found, and it indicates a corrupt database
00053         // file
