00003 /**
00004  * @ingroup tablet
00005  * @brief Add a lock to a tablet
00006  *
00007  * Add a lock on a certain tablet to ensure that it remains in that tablet slot.
00008  * Locks accumulate; all slots start with a status of 0, which is raised to 1
00009  * when a tablet is loaded into it. A lock increments this number, so a tablet
00010  * with a status of 2 or above can not be removed from that slot, even when the
00011  * database is closed. Thus, all locks held on a tablet must be released before
00012  * closing the tablet's database. If the tablet cannot be found to be locked,
00013  * then return VIRG_FAIL.
00014  *
00015  * @param v     Pointer to the state struct of the database system
00016  * @param id    ID of the tablet to be locked
00017  * @return VIRG_SUCCESS or VIRG_FAIL depending on errors during the function
00018  * call
00019  */
00020 int virg_tablet_lock(virginian *v, unsigned tablet_id)
00021 {
00022 #ifdef VIRG_DEBUG_SLOTS
00023         fprintf(stderr, "trying to lock %i\n", tablet_id);
00024         virg_print_slots(v);
00025 #endif
00026 
00027         // tablet slot lock
00028         pthread_mutex_lock(&v->slot_lock);
00029 
00030         // look for tablet to lock
00031         unsigned i;
00032         for(i = 0; i < VIRG_MEM_TABLETS; i++)
00033                 if(v->tablet_slot_status[i] > 0 && v->tablet_slot_ids[i] == tablet_id)
00034                         break;
00035 
00036         // check to make sure the tablet was found
00037 #ifdef VIRG_DEBUG
00038         if(i == VIRG_MEM_TABLETS) {
00039                 virg_print_slots(v);
00040                 fprintf(stderr, "looking for %u\n", tablet_id);
00041         }
00042 #endif
00043         VIRG_DEBUG_CHECK(i == VIRG_MEM_TABLETS, "Couldn't find tablet to lock");
00044 
00045         // increase the number of read locks
00046         v->tablet_slot_status[i]++;
00047 
00048         // unlock tablet slots
00049         pthread_mutex_unlock(&v->slot_lock);
00050 
00051         return VIRG_SUCCESS;
00052 }
