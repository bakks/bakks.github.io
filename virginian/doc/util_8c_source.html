00056 /**
00057  * @brief Returns the time since the timer's start
00058  *
00059  * Subtracts the current value returned by gettimeofday() to that set in the
00060  * virg_timer_start() function. Note that this doesn't actually stop the timer,
00061  * as it does not change the value of virg_starttime.
00062  *
00063  * @return The time in seconds since virg_timer_start() was called
00064  */
00065 double virg_timer_stop()
00066 {
00067         gettimeofday(&virg_endtime, NULL);
00068         double start = virg_starttime.tv_sec +
00069                 (double)virg_starttime.tv_usec * .000001;
00070         double end = virg_endtime.tv_sec +
00071                 (double)virg_endtime.tv_usec * .000001;
00072         return (end - start);
00073 }
