00010  * @ingroup vm
00011  * @brief Chooses and prepares for execution on one or multiple CPU cores
00012  *
00013  * This function is intended to make the virg_vm_execute() function by making
00014  * the choice between executing with a single or multiple CPU cores transparent
00015  * based on the virginian.use_multi. If this value is false, we loop and call
00016  * virginia_single for every tablet to be processes. Otherwise, we create
00017  * virginian.multi_threads threads which greedily process as many data tablets
00018  * as they can and wait for them to finish before returning. If num_tablets is
00019  * 0, then there is no restriction on how many data tablets will be processed in
00020  * this function.
00021  *
00022  * @param v             Pointer to the state struct of the database system
00023  * @param vm    Pointer to the context struct of the virtual machine
00024  * @param tab   Pointer to the pointer to the current data tablet to process
00025  * @param res   Pointer to the pointer to the current result tablet
00026  * @param num_tablets The number of tablets to process during this call, 0 for
00027  * as many as possible
00028  * @return VIRG_SUCCESS or VIRG_FAIL depending on errors during the function
00029  * call
00030  */
00031 int virg_vm_cpu(virginian *v, virg_vm *vm, virg_tablet_meta **tab, virg_tablet_meta **res, unsigned num_tablets)
00032 {
00033         unsigned proced = 0;
00034 
00035         // if the virginian struct is set to execute using only a single core
00036         if(!v->use_multi) {
00037                 // infinite loop
00038                 while(1) {
00039                         // add an extra lock to the current data and result tablets that
00040                         // will be released at a lower level within virginia_single()
00041                         virg_tablet_lock(v, tab[0]->id);
00042                         virg_tablet_lock(v, res[0]->id);
00043 
00044                         // single core execution function
00045                         virginia_single(v, vm, tab[0], res, 0, tab[0]->rows);
00046 
00047                         // we've processed another tablet
00048                         proced++;
00049 
00050                         // break out if we've processed enough tablets
00051                         if((num_tablets != 0 && proced >= num_tablets) || tab[0]->last_tablet)
00052                                 break;
00053 
00054                         // load the next data tablet
00055                         virg_db_loadnext(v, tab);
00056                 }
00057         }
00058         else {
00059                 pthread_t thread[v->multi_threads];
00060 
00061                 // copy values into the argument structure that must be used with
00062                 // pthread_create()
00063                 virg_vm_arg arg;
00064                 arg.v = v;
00065                 arg.vm = vm;
00066                 arg.tab = tab[0];
00067                 arg.res = res[0];
00068                 arg.row = 0;
00069                 arg.num_rows = 0;
00070                 arg.num_tablets = num_tablets;
00071                 arg.tablets_proced = 0;
00072 
00073                 // mutexes for dealing with the current last tablet and result tablets
00074                 VIRG_CHECK(pthread_mutex_init(&arg.tab_lock, NULL), "Could not init mutex")
00075                 VIRG_CHECK(pthread_mutex_init(&arg.res_lock, NULL), "Could not init mutex")
00076 
00077                 // create tablet processing threads
00078                 unsigned i;
00079                 for(i = 0; i < v->multi_threads; i++)
00080                         pthread_create(&thread[i], NULL, virginia_multi, (void*) &arg);
00081 
00082                 // wait for all threads to run out of data to process and finish
00083                 for(i = 0; i < v->multi_threads; i++)
00084                         pthread_join(thread[i], NULL);
00085 
00086                 tab[0] = arg.tab;
00087                 res[0] = arg.res;
00088 
00089                 VIRG_CHECK(pthread_mutex_destroy(&arg.res_lock), "Could not destroy mutex")
00090                 VIRG_CHECK(pthread_mutex_destroy(&arg.tab_lock), "Could not destroy mutex")
00091         }
00092
