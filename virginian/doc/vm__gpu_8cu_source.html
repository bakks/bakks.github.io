00053 /**
00054  * @ingroup vm
00055  * @brief Execute the data-parallel portion of an opcode program on a GPU
00056  *
00057  * This function handles setting up and launching the virtual machine kernel
00058  * that executes the virtual machine on the GPU. The user has a choice between
00059  * serial kernel executions for each tablet, streaming executions with
00060  * overlapping memcpys, and mapped execution, which is currently the fastest
00061  * option.
00062  *
00063  * If streaming and memory mapping are both disabled in the virginian struct,
00064  * then kernels will be processed on the GPU serially. This means that the data
00065  * tablet will be transferred to GPU memory, the virtual machine executed, then
00066  * the result tablet transferred back for each tablet with no overlap. This is
00067  * the slowest method of GPU execution, and it does not require pinned memory.
00068  *
00069  * If streaming is enabled, regardless of the mapped memory setting, then it is
00070  * used. This works by allocating a fixed number of tablet streams, set equal to
00071  * half of the number of allocated GPU tablet slots, and overlapping data
00072  * transfer, kernel execution, and result transfer for tablets. Each stream gets
00073  * a data and result tablet slot on the GPU, and the loop iterates through each
00074  * stream in a round-robin fashion executing virtual machines on tablets. Note
00075  * that when we loop back and re-use streams then we block to wait for execution
00076  * if the previous asynchronous launches. On Tesla C1060 hardware you cannot
00077  * overlap transfers to the GPU with transfers from it, and thus streaming is
00078  * about as fast as serial execution. Also note that on this hardware the
00079  * semantics for asynchronous operations are somewhat frusterating. Based on
00080  * informal tests, it appears that even though asynchronous operations are
00081  * completely asynchronous with respect to the calling thread, the order in
00082  * which they are added to streams has an effect on when they are run. Thus,
00083  * there are non-optimal orderings of memory copies and kernel executions when
00084  * multiple asynchronous operations are queued in multiple streams. Even with
00085  * smarter ordering, however, the inability of current hardware to transfer both
00086  * ways across the PCI bus simultaneously means that this method of execution is
00087  * currently only useful if either the data or results of a query are resident
00088  * in GPU memory for the entire processing operation. For some reason I could
00089  * only create 6 streams at a time during my testing, so you may encounter
00090  * problems if you allocate more than 12 gpu tablet slots.
00091  *
00092  * Mapped memory is currently the fastest method for GPU processing by far. It
00093  * is also much simpler, since we have fewer memory copies and we don't have to
00094  * manage multiple streams at the same time. The tablet row counter is
00095  * implemented as a separate variable for mapped memory, since it would be very
00096  * very expensive to perform atomic operations on a mapped location.
00097  *
00098  * @param v     Pointer to the state struct of the database system
00099  * @param vm    Pointer to the context struct of the virtual machine
00100  * @param tab   Pointer to the pointer to the current data tablet to process
00101  * @param res   Pointer to the pointer to the current result tablet
00102  * @param num_tablets Number of tablets to process on the GPU, 0 if as many as
00103  * possible
00104  * @return VIRG_SUCCESS or VIRG_FAIL depending on errors during the function
00105  * call
00106  */

00544                         //fprintf(stderr, "block %u: %f %f %f %f\n", proced, cum, f[0], f[1], f[2]);
00545                         
00546                         // add to cumulative time
00547                         cum += f[0] + f[1] + f[2];
00548 
00549                         proced++;
00550 
00551                         // check if we're done processing tablets
00552                         if(tab[0]->last_tablet || (num_tablets != 0 && proced >= num_tablets))
00553                                 break;
00554 
00555                         // load next data tablet
00556                         virg_db_loadnext(v, tab);
00557 
00558                         // if this data tablet has no rows, finish
00559                         if(tab[0]->rows == 0)
00560                                 break;
00561 
00562                         virg_tablet_meta *temp = res[0];
00563                         virg_vm_allocresult(v, vm_, res, res[0]);
00564                         virg_tablet_unlock(v, temp->id);
00565                 }
00566 
00567                 // destruct timers
00568                 cudaEventDestroy(start);
00569                 cudaEventDestroy(data);
00570                 cudaEventDestroy(kernel);
00571                 cudaEventDestroy(results);
00572         }
00573 
00574         // wait for all cuda operations to finish
00575         cudaDeviceSynchronize();
00576 
00577         return VIRG_SUCCESS;
00578 }

00893 __device__ __forceinline__ void op_Result OPARGS
00894 {
00895         /**
00896          * To manage outputting result rows, every thread atomically increments a
00897          * shared variable to determine how many result rows there will be, then the
00898          * first thread in the block atomically adds this number to a global
00899          * variable, with appropriate threadblock synchronization in between these
00900          * calls. This is more efficient than a scan operation in this case, since
00901          * we don't care about the order of the rows and shared memory atomic
00902          * operations are actually fairly efficient when compared to reduction
00903          * operations, which tend to cause shared memory bank conflicts.
00904          */
00905         unsigned place;
00906 
00907         __syncthreads();
00908 
00909         // if this is a valid row, update the shared variable
00910         if(valid)
00911                 place = atomicAdd(&block, 1);
00912 
00913         __syncthreads();
00914 
00915         unsigned num_rows = block;
00916 
00917         // the first thread in the block updates the global variable
00918         if(threadIdx.x == 0) {
00919                 if(scratch != NULL) // mapped
00920                         bstart = atomicAdd(&row_counter, block);
00921                 else {
00922                         bstart = atomicAdd(&res->rows, block);
00923                 }
00924         }
00925 
00926         __syncthreads();
00927 
00928         // TODO check for result tablet overflow
00929 
00930         unsigned block_start = bstart;
00931         //block = 0;
00932         unsigned j;
00933         char *p;
00934 
00935         // if this is a result row
00936         if(valid)
00937                 // for every register to be output for this result row
00938                 for(j = op.p1; j < op.p1 + op.p2; j++) {
00939                         // register/column stride
00940                         unsigned stride = context.stride[j];
00941                         unsigned col_location = stride * (block_start + place);
00942 
00943                         // if not mapped, write to the result tablet
00944                         if(scratch == NULL || VIRG_NOTWOSTEP)
00945                                 p = (char*)res + meta_res->fixed_block +
00946                                         meta_res->fixed_offset[j - op.p1] + col_location;
00947 
00948                         // if mapped, write to the scratch memory area for buffering before
00949                         // sending across the PCI bus
00950                         else
00951                                 p = (char*)scratch + meta_res->fixed_block +
00952                                         meta_res->fixed_offset[j - op.p1] + col_location;
00953 
00954                         //printf("write row %u\n", place);
00955 
00956                         // switch the write based on the variable stride
00957                         switch(stride) {
00958                                 case 4:
00959                                         ((int*)p)[0] = context.reg[j].i;
00960                                         break;
00961                                 case 8:
00962                                         ((double*)p)[0] = context.reg[j].d;
00963                                         break;
00964                                 case 1:
00965                                         p[0] = context.reg[j].c;
00966                                         break;
00967                         }
00968                 }
00969 
00970 
00971         /**
00972          * If mapped memory is being used, result rows are written back to main
00973          * memory in a two step process. This both reduces the number of accesses
00974          * that have to cross the PCI bus, and ensures that the results are being
00975          * coalesced properly, as the coalescing rules for mapped memory appear to
00976          * be slightly more strict than for GPU global memory. First, result rows
00977          * are written to GPU global memory, exactly how they would be if serial
00978          * execution and memory transfers were being used. The results are divided
00979          * into threadblock-sized blocks, and once we are sure that the results are
00980          * written to global memory, we increment the global counters for the blocks
00981          * that we have written to. After incrementing, we check to see if the
00982          * blocks we have written to are completely filled with rows. If so, every
00983          * thread in the threadblock transfers a row from the result block to main
00984          * memory, thus maximizing efficiency and coalescing. Tests show that this
00985          * is a very good way of managing result transfers of this kind, since
00986          * transfers back to main memory proceed efficiently but are overlapped with
00987          * kernel execution.
00988          */
00989 
00990         // if we are using mapped memory
00991         if(scratch != NULL && !VIRG_NOTWOSTEP)
00992         {
00993                 // make sure our result writes have reached global memory
00994                 __threadfence();
00995                 __syncthreads();
00996 
00997                 // only do this in the first cuda thread
00998                 if(threadIdx.x == 0) {
00999                         shared_blockorder = atomicAdd(&threadblock_order, 1);
01000                         
01001                         if(num_rows > 0) {
01002                                 unsigned *threadswritten = (unsigned*)((char*)scratch +
01003                                         VIRG_TABLET_SIZE);
01004                                 unsigned result_blockid = block_start / VIRG_THREADSPERBLOCK;//blockDim.x;
01005 
01006                                 unsigned blockbreak = ((block_start + num_rows - 1) &
01007                                         VIRG_THREADSPERBLOCK_MASK);
01008                                 unsigned x, y;
01009                         
01010                                 // if the results span over two different thread block areas
01011                                 if((blockbreak > (block_start & VIRG_THREADSPERBLOCK_MASK)) &&
01012                                         (block_start + num_rows != blockbreak))
01013                                 {
01014                                         // increment the rows that have been written for both
01015                                         unsigned thisnewrows = blockbreak - block_start;
01016                                         x = atomicAdd(&threadswritten[result_blockid], thisnewrows);
01017                                         thisblockwritten = x + thisnewrows;
01018 
01019                                         unsigned nextnewrows = block_start + num_rows - blockbreak;
01020                                         y = atomicAdd(&threadswritten[result_blockid + 1],
01021                                                 nextnewrows);
01022                                         nextblockwritten = y + nextnewrows;
01023                                 }
01024                                 // otherwise increment the rows that have been written for only
01025                                 // this result threadblock area
01026                                 else {
01027                                         x = atomicAdd(&threadswritten[result_blockid], num_rows);
01028                                         thisblockwritten = x + num_rows;
01029                                         nextblockwritten = 0;
01030                                 }
01031                         }
01032                 }
01033 
01034                 __syncthreads();
01035         
01036                 // if an entire threadblock-sized area has been filled with result rows
01037                 if(num_rows > 0 && thisblockwritten == VIRG_THREADSPERBLOCK)
01038                 {
01039                         unsigned aligned_start = block_start & VIRG_THREADSPERBLOCK_MASK;
01040 
01041                         // do coalesced writes from global memory to mapped main memory of
01042                         // this block of results
01043                         for(j = op.p1; j < op.p1 + op.p2; j++) {
01044                                 unsigned stride = context.stride[j];
01045 
01046                                 p = (char*)res + meta_res->fixed_block +
01047                                         meta_res->fixed_offset[j - op.p1] + stride * (aligned_start + threadIdx.x);
01048                                 char *p_src = (char*)scratch + meta_res->fixed_block +
01049                                         meta_res->fixed_offset[j - op.p1] + stride * (aligned_start + threadIdx.x);
01050 
01051                                 switch(stride) {
01052                                         case 4:
01053                                                 ((int*)p)[0] = ((int*)p_src)[0];
01054                                                 break;
01055                                         case 8:
01056                                                 ((int*)p)[0] = ((int*)p_src)[0];
01057                                                 break;
01058                                         case 1:
01059                                                 if(threadIdx.x < VIRG_THREADSPERBLOCK / 4)
01060                                                         ((int*)p)[0] = ((int*)p_src)[0];
01061                                                 break;
01062                                 }
01063                         }
01064                 }
01065 
01066                 if(shared_blockorder == gridDim.x - 1)
01067                 {
01068                         unsigned aligned_start = row_counter & VIRG_THREADSPERBLOCK_MASK;
01069 
01070                         // do coalesced writes from global memory to mapped main memory of
01071                         // this block of results
01072                         for(j = op.p1; j < op.p1 + op.p2; j++) {
01073                                 unsigned stride = context.stride[j];
01074 
01075                                 p = (char*)res + meta_res->fixed_block +
01076                                         meta_res->fixed_offset[j - op.p1] + stride * (aligned_start + threadIdx.x);
01077                                 char *p_src = (char*)scratch + meta_res->fixed_block +
01078                                         meta_res->fixed_offset[j - op.p1] + stride * (aligned_start + threadIdx.x);
01079 
01080                                 switch(stride) {
01081                                         case 4:
01082                                                 ((int*)p)[0] = ((int*)p_src)[0];
01083                                                 break;
01084                                         case 8:
01085                                                 ((int*)p)[0] = ((int*)p_src)[0];
01086                                                 break;
01087                                         case 1:
01088                                                 if(threadIdx.x < VIRG_THREADSPERBLOCK / 4)
01089                                                         ((int*)p)[0] = ((int*)p_src)[0];
01090                                                 break;
01091                                 }
01092                         }
01093                 }
01094 
01095                 // if a second thread-block sized area has been filled with result rows
01096                 if(num_rows > 0 && nextblockwritten == VIRG_THREADSPERBLOCK)
01097                 {
01098                         unsigned aligned_start = (block_start + num_rows) & VIRG_THREADSPERBLOCK_MASK;
01099                         // do coalesced writes from global memory to mapped main memory of
01100                         // this block of results
01101                         for(j = op.p1; j < op.p1 + op.p2; j++) {
